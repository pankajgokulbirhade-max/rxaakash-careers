function escapeHtml(text) {
    if (!text) return '';
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}

function performSearch() {
    const query = document.getElementById('searchInput').value.toLowerCase().trim();
    const location = document.getElementById('locationSelect').value;
    const dept = document.getElementById('deptSelect').value;

    let results = allJobs.filter(job => {
        const matchQuery = !query ||
            job.title.toLowerCase().includes(query) ||
            job.company.toLowerCase().includes(query) ||
            job.dept.toLowerCase().includes(query) ||
            job.tags.some(t => t.toLowerCase().includes(query));

        const matchLocation = location === 'all' ||
            job.location.toLowerCase().includes(location.replace(/-/g, ' '));

        const matchDept = dept === 'all' ||
            job.dept.toLowerCase().includes(dept.replace(/-/g, ' '));

        return matchQuery && matchLocation && matchDept;
    });

    const activeFilters = document.querySelectorAll('.filter-tag.active[data-filter]');
    activeFilters.forEach(tag => {
        const f = tag.dataset.filter;
        if (f === 'remote') results = results.filter(j => j.remote);
        if (f === 'fresher') results = results.filter(j => j.level === 'Entry');
        if (f === 'experienced') results = results.filter(j => j.level !== 'Entry');
        if (f === 'urgent') results = results.filter(j => j.urgent);
    });

    filteredJobs = results;
    sortJobs(currentSort);
}

function sortJobs(value) {
    if (value) currentSort = value;
    const sorted = [...filteredJobs];
    if (currentSort === 'salary-high') {
        sorted.sort((a, b) => {
            const n = s => parseInt(s.replace(/[₹,\s\/a-z]+/gi, '').split('-')[0]) || 0;
            return n(b.salary) - n(a.salary);
        });
    } else {
        sorted.sort((a, b) => {
            const d = p => parseInt(p.match(/\d+/)?.[0] || 99);
            return d(a.posted) - d(b.posted);
        });
    }
    renderJobs(sorted);
}
